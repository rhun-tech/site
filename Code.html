What's neat is loading the routes and clusters definitions from memory using LoadFromMemory() as opposed to LoadFromConfig(). 

services.AddReverseProxy()
  .LoadFromMemory(routes, clusters)

Then add your route:

var routeConfig = new RouteConfig
  {
      RouteId = "SampleRoute",
      Match = new RouteMatch
      {
          // Host to match
          Hosts = new[] { "example.com" },
          // Path to match
          Path = "/api", 
      },
      // Cluster to forward to
      ClusterId = "SampleCluster", 
  };

Finally add your cluster definition:

var clusterConfig = new ClusterConfig
  {
      ClusterId = "SampleCluster",
      Destinations = new Dictionary
      {
          // Backend server address
          ["backend1"] = new DestinationConfig { Address = "http://backend1:5000" },
          // Another backend server address
          ["backend2"] = new DestinationConfig { Address = "http://backend2:5000" },
      },
  };

Then you can access your API through the reverse proxy http://example.com/api/{resource} where {resource} is your desired endpoint.

Each section on this page has it's own reference tag to the title-section-component.

Html

<title-section-component title="Template">
</title-section-component>

Javascript

class TitleSection extends HTMLElement {
  constructor() {
      super();
  }

  connectedCallback() {
      const titleText = this.getAttribute('title') || 'Default Title';
      const titleIcon = this.getAttribute('icon') || '';

      this.innerHTML = `
        <div class="hstack pb-2">
        <div id="title" class="h4">${titleText}</div>
        <span class="iconify rhun-2 ms-auto" data-icon="${titleIcon}"></span>
        </div>
        <hr />
      `;
  }
}
            
customElements.define('title-section-component', TitleSection);
